#include "esp_camera.h"
#include <WiFi.h>
#include "esp_http_server.h" // 需要包含这个库来创建网页服务器
// 替换为您的 Wi-Fi 名称和密码
const char* ssid = "";
const char* password = "";

// #define PWDN_GPIO_NUM      5  // OV_PWDN 
// #define RESET_GPIO_NUM     4  // OV_RESET 
// #define XCLK_GPIO_NUM      -1  // 这个引脚需要手动连接到模块的 XCLK 输入点
// #define SIOD_GPIO_NUM      2  // OV_SDA 
// #define SIOC_GPIO_NUM      1  // OV_SCL 

// #define Y9_GPIO_NUM        9  // OV_D7 
// #define Y8_GPIO_NUM        10 // OV_D6 
// #define Y7_GPIO_NUM        11 // OV_D5 
// #define Y6_GPIO_NUM        12 // OV_D4 
// #define Y5_GPIO_NUM        14 // OV_D3 
// #define Y4_GPIO_NUM        13 // OV_D2 
// #define Y3_GPIO_NUM        16 // OV_D1 
// #define Y2_GPIO_NUM        15 // OV_D0

// #define HREF_GPIO_NUM      7  // OV_HREF

#define PWDN_GPIO_NUM      5
#define RESET_GPIO_NUM     4
#define XCLK_GPIO_NUM      -1  // -1 表示模块自带时钟，ESP32不提供
#define SIOD_GPIO_NUM      2
#define SIOC_GPIO_NUM      1

#define Y9_GPIO_NUM        9
#define Y8_GPIO_NUM        10
#define Y7_GPIO_NUM        11
#define Y6_GPIO_NUM        12
#define Y5_GPIO_NUM        14
#define Y4_GPIO_NUM        13
#define Y3_GPIO_NUM        16
#define Y2_GPIO_NUM        15

#define VSYNC_GPIO_NUM     6
#define HREF_GPIO_NUM      7
#define PCLK_GPIO_NUM      8

static esp_err_t stream_handler(httpd_req_t *req){
  camera_fb_t * fb = NULL;
  esp_err_t res = ESP_OK;
  size_t _jpg_buf_len = 0;
  uint8_t * _jpg_buf = NULL;
  char * part_buf[64];

  res = httpd_resp_set_type(req, "multipart/x-mixed-replace;boundary=--frame");
  if(res != ESP_OK){
    return res;
  }

  while(true){
    fb = esp_camera_fb_get();
    if (!fb) {
      Serial.println("Camera capture failed");
      res = ESP_FAIL;
    } else {
      if(fb->format != PIXFORMAT_JPEG){
        bool jpeg_converted = frame2jpg(fb, 80, &_jpg_buf, &_jpg_buf_len);
        esp_camera_fb_return(fb);
        fb = NULL;
        if(!jpeg_converted){
          Serial.println("JPEG compression failed");
          res = ESP_FAIL;
        }
      } else {
        _jpg_buf_len = fb->len;
        _jpg_buf = fb->buf;
      }
    }
    if(res == ESP_OK){
      size_t hlen = snprintf((char *)part_buf, 64, "Content-Type: image/jpeg\r\nContent-Length: %u\r\n\r\n", _jpg_buf_len);
      res = httpd_resp_send_chunk(req, (const char *)part_buf, hlen);
    }
    if(res == ESP_OK){
      res = httpd_resp_send_chunk(req, (const char *)_jpg_buf, _jpg_buf_len);
    }
    if(res == ESP_OK){
      res = httpd_resp_send_chunk(req, "\r\n--frame\r\n", strlen("\r\n--frame\r\n"));
    }
    if(fb){
      esp_camera_fb_return(fb);
      fb = NULL;
      _jpg_buf = NULL;
    } else if(_jpg_buf){
      free(_jpg_buf);
      _jpg_buf = NULL;
    }
    if(res != ESP_OK){
      break;
    }
  }
  return res;
}

void startCameraServer(){
  httpd_handle_t stream_httpd = NULL;
  httpd_config_t config = HTTPD_DEFAULT_CONFIG();

  httpd_uri_t stream_uri = {
    .uri       = "/",
    .method    = HTTP_GET,
    .handler   = stream_handler,
    .user_ctx  = NULL
  };

  if (httpd_start(&stream_httpd, &config) == ESP_OK) {
    httpd_register_uri_handler(stream_httpd, &stream_uri);
  }
}

// ===================================================
// 主程序
// ===================================================

void setup() {
  Serial.begin(115200);
  Serial.setDebugOutput(true);
  Serial.println();

  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sccb_sda = SIOD_GPIO_NUM;
  config.pin_sccb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  // config.xclk_freq_hz = 0; // 当pin_xclk为-1时，此项无效
  config.xclk_freq_hz = 20000000; // 提供一个非零的虚拟值，例如20MHz
  config.pixel_format = PIXFORMAT_JPEG;

  // 对于高分辨率 (OV5640 支持 UXGA, QXGA 等)
  // 您可能需要使用 PSRAM。请确保您的 ESP32-S3 开发板有 PSRAM。
  config.frame_size = FRAMESIZE_UXGA; // 1600x1200
  config.jpeg_quality = 12; // 质量可以适当调整 (0-63, 数值越小质量越高)
  config.fb_count = 2;
  config.fb_location = CAMERA_FB_IN_PSRAM; // 强制使用PSRAM

  // 初始化摄像头
  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("摄像头初始化失败，错误码: 0x%x", err);
    return;
  }
  Serial.println("摄像头初始化成功！");

  // 连接 WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi 已连接");
  Serial.print("摄像头视频流地址: http://");
  Serial.println(WiFi.localIP());

  // 启动网页服务器
  startCameraServer();
}

void loop() {
  // 主循环保持空闲，因为服务器在后台运行
  delay(10000);
}
