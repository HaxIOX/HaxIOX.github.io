浏览器脚本

  1. JavaScript 脚本：
    • 执行环境：通常嵌入在网页中，通过 <script> 标签加载，或者在浏览器的控制台中运行。
    • 目的：实现页面功能、处理用户交互、操作 DOM（文档对象模型）等。
  2. 用户脚本：
    • 执行环境：由用户脚本管理器（如 Tampermonkey 或 Greasemonkey）注入到网页中。它们运行在用户的浏览器中，可以修改网页的行为和外观。
    • 目的：用户定制网页的功能，添加额外的功能或修复网页上的问题。

document.addEventListener

document.addEventListener 是 JavaScript 的标准方法，用于在指定的事件发生时执行指定的函数。这是 DOM API 的一部分，允许你在网页上监听和处理用户交互或其他事件。

用法示例：

document.addEventListener('DOMContentLoaded', function() {
    // 代码在文档完全加载后运行
    console.log('页面加载完成');
});

Node.js 环境

如果你在 Node.js 环境中运行 JavaScript 代码，document 对象是不存在的，因为 Node.js 是一个服务器端的 JavaScript 环境，没有浏览器提供的 DOM API。

解决方法：

  • 如果你需要在 Node.js 中处理网页内容，考虑使用类似 jsdom 的库来模拟浏览器环境。
javascript复制代码const { JSDOM } = require('jsdom');
const dom = new JSDOM('<html><body><h1>Hello world</h1></body></html>');
const document = dom.window.document;

// 现在你可以使用 document 对象
console.log(document.querySelector('h1').textContent); // 输出: Hello world

// ==UserScript==
// @name         百度网盘提取码助手（终极版 v2.5）
// @namespace    https://yourdomain.dev
// @version      2.5
// @description  选中链接+码→跳转 #pwd→轮询填码→美观提示（已支持 share/init?surl & SPA）
// @match        *://*/*
// @match        *://pan.baidu.com/*
// @grant        none
// @run-at       document-end
// ==/UserScript==

;(function(){
  'use strict';

  // —— 正则 ——
  const SHARE_RE = /https?:\/\/pan\.baidu\.com\/s\/([A-Za-z0-9\-_]+)/;
  const CODE_RE  = /(?:提取码|密码)[:：]?\s*([A-Za-z0-9]{4})/;
  const HASH_RE  = /[#&]pwd=([A-Za-z0-9]{4})/;

  // —— 浮层提示 ——
  function showTip(msg) {
    const d = document.createElement('div');
    d.textContent = msg;
    Object.assign(d.style, {
      position:   'fixed',
      top:        '20px',
      right:      '20px',
      padding:    '10px 15px',
      background: 'rgba(0,0,0,0.75) !important',
      color:      '#fff',
      fontSize:   '14px',
      borderRadius:'6px',
      boxShadow:  '0 4px 12px rgba(0,0,0,0.3)',
      zIndex:     '2147483647',
      opacity:    '1',
      transition: 'opacity 0.4s',
      pointerEvents:'none'
    });
    document.documentElement.appendChild(d);
    setTimeout(() => d.style.opacity = '0', 2000);
    setTimeout(() => d.remove(), 2600);
  }

  // —— 1. 非 pan.baidu.com 页面：选中跳转+Hash ——
  if (!/pan\.baidu\.com/.test(location.hostname)) {
    document.addEventListener('mouseup', () => {
      const txt = window.getSelection().toString().trim();
      const sm = txt.match(SHARE_RE);
      if (!sm) return;
      const shareId = sm[1];
      const cm = txt.match(CODE_RE);
      showTip('✅ 检测到百度网盘链接');
      if (cm) {
        const pwd = cm[1];
        if (confirm(`链接：\nhttps://pan.baidu.com/s/${shareId}\n提取码：${pwd}\n\n跳转并自动填码？`)) {
          window.open(`https://pan.baidu.com/s/${shareId}#pwd=${pwd}`, '_blank');
        }
      } else {
        if (confirm(`链接：\nhttps://pan.baidu.com/s/${shareId}\n无提取码，是否跳转？`)) {
          window.open(`https://pan.baidu.com/s/${shareId}`, '_blank');
        }
      }
    });
    return;
  }

  // —— 2. pan.baidu.com 页面：从 Hash 读码并自动填 ——
  const hm = location.hash.match(HASH_RE);
  if (!hm) return;
  const code = hm[1];
  showTip('检测到提取码：' + code);

  // —— 3. 轮询尝试填写 ——
  let attempts = 0;
  const maxAttempts = 20;
  const timer = setInterval(() => {
    attempts++;
    // 输入框
    const inp = document.querySelector(
      '#accessCode, .pickpw input, input.QKKaIE, input[placeholder*="提取码"]'
    );
    // 按钮
    const btn = document.querySelector('#submitBtn, .submit-btn-text')
             || Array.from(document.querySelectorAll('button, a, div'))
                     .find(el => /提取文件|提交/.test(el.innerText.trim()));

    if (inp && btn) {
      inp.focus();
      inp.value = code;
      inp.dispatchEvent(new Event('input', { bubbles: true }));
      btn.click();
      showTip('✅ 已自动填写并提交');
      clearInterval(timer);
    } else if (attempts >= maxAttempts) {
      clearInterval(timer);
      showTip('⚠️ 未能自动填码，请手动输入');
    }
  }, 500);

})();

    // 提取百度网盘链接信息
    function extractBaiduLinks(text) {
        // 更宽松的链接匹配
        const linkRegex = /https?:\/\/pan\.baidu\.com\/s\/[a-zA-Z0-9_\-]+(\?pwd=[a-zA-Z0-9]{4})?/gi;
        const links = [...text.matchAll(linkRegex)];

        if (links.length === 0) return null;

        const results = [];

        for (const linkMatch of links) {
            const fullLink = linkMatch[0];
            const [baseUrl, pwdParam] = fullLink.split('?pwd=');

            let extractCode = pwdParam || null;

            // 如果链接中没有提取码，尝试从周围文本中查找
            if (!extractCode) {
                // 查找链接附近的提取码
                const linkIndex = text.indexOf(fullLink);
                const contextStart = Math.max(0, linkIndex - 100);
                const contextEnd = Math.min(text.length, linkIndex + fullLink.length + 100);
                const context = text.substring(contextStart, contextEnd);

                const codePatterns = [
                    /(?:提取码|密码|验证码|码|pwd|password)[:：\s]*([a-zA-Z0-9]{4})/i,
                    /\b([a-zA-Z0-9]{4})\b/g
                ];

                for (const pattern of codePatterns) {
                    const matches = [...context.matchAll(pattern)];
                    for (const match of matches) {
                        const code = match[1];
                        // 验证提取码格式（4位字母数字组合）
                        if (/^[a-zA-Z0-9]{4}$/.test(code) && code !== fullLink.slice(-4)) {
                            extractCode = code;
                            break;
                        }
                    }
                    if (extractCode) break;
                }
            }

            results.push({
                link: baseUrl,
                extractCode: extractCode,
                fullLink: fullLink
            });
        }

        return results.length > 0 ? results : null;
    }

    // 显示弹窗
    function showDialog(linkInfos) {

        const existingOverlay = document.querySelector('.baidu-helper-overlay');
        if (existingOverlay) {
            existingOverlay.remove();

        injectStyles();

        // 创建遮罩层
        const overlay = document.createElement('div');
        overlay.className = 'baidu-helper-overlay';
        document.body.appendChild(overlay);

        // 创建弹窗
        const dialog = document.createElement('div');
        dialog.className = 'baidu-helper-dialog';

        // 头部
        const header = document.createElement('div');
        header.className = 'baidu-helper-header';
        header.innerHTML = `
            <div class="baidu-helper-icon">
                <svg width="16" height="16" viewBox="0 0 24 24" fill="white">
                    <path d="M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2zm-2 15l-5-5 1.41-1.41L10 14.17l7.59-7.59L19 8l-9 9z"/>
                </svg>
            </div>
            <h3 class="baidu-helper-title">✨ 发现百度网盘链接</h3>
        `;

        // 内容
        const content = document.createElement('div');
        content.className = 'baidu-helper-content';

        linkInfos.forEach((info, index) => {
            const item = document.createElement('div');
            item.className = 'baidu-helper-link-item';

            let itemHTML = `
                <div class="baidu-helper-label">🔗 链接 ${index + 1}:</div>

            if (info.extractCode) {

                    <div class="baidu-helper-text">${info.extractCode}</div>

            }

            item.innerHTML = itemHTML;
            content.appendChild(item);
        });

        // 按钮
        const actions = document.createElement('div');
        actions.className = 'baidu-helper-actions';

        const cancelBtn = document.createElement('button');
        cancelBtn.className = 'baidu-helper-btn baidu-helper-btn-secondary';
        cancelBtn.textContent = '取消';

        const confirmBtn = document.createElement('button');
        confirmBtn.className = 'baidu-helper-btn baidu-helper-btn-primary';
        confirmBtn.textContent = linkInfos.length === 1 ?
            (linkInfos[0].extractCode ? '🚀跳转并填写提取码' : '🚀跳转到链接') :
            '🚀打开所有链接';

        actions.appendChild(cancelBtn);
        actions.appendChild(confirmBtn);

        dialog.appendChild(header);
        dialog.appendChild(content);
        dialog.appendChild(actions);
        overlay.appendChild(dialog);

        // 事件处理
        const closeDialog = () => overlay.remove();

        cancelBtn.onclick = closeDialog;
        overlay.onclick = (e) => {
            if (e.target === overlay) closeDialog();

        confirmBtn.onclick = () => {
            linkInfos.forEach(info => {
                if (info.extractCode) {
                    sessionStorage.setItem('baiduAutoFillCode_' + Date.now(), info.extractCode);
                    const targetUrl = info.link + '?pwd=' + info.extractCode;
                    window.open(targetUrl, '_blank');

                    window.open(info.link, '_blank');

            closeDialog();

    }

    // 文本选择监听
    function handleTextSelection() {
        let selectionTimeout;

        document.addEventListener('mouseup', () => {
            clearTimeout(selectionTimeout);
            selectionTimeout = setTimeout(() => {
                const selection = window.getSelection();
                const selectedText = selection.toString().trim();

                if (selectedText.length < 20) return;

                console.log('百度网盘助手: 检测选中文本:', selectedText.substring(0, 100) + '...');

                const linkInfos = extractBaiduLinks(selectedText);
                if (linkInfos) {
                    console.log('百度网盘助手: 发现链接:', linkInfos);
                    showDialog(linkInfos);
                    selection.removeAllRanges(); // 清除选择
                }
            }, 200);
        });
    }

    // 自动填写提取码
    function autoFillExtractCode() {
        // 查找sessionStorage中的提取码
        const codes = [];
        for (let i = 0; i < sessionStorage.length; i++) {
            const key = sessionStorage.key(i);
            if (key && key.startsWith('baiduAutoFillCode_')) {
                codes.push({
                    key: key,
                    code: sessionStorage.getItem(key)
                });
            }
        }

        if (codes.length === 0) return;

        console.log('百度网盘助手: 找到提取码', codes);

        // 常见的输入框选择器
        const inputSelectors = [
            'input[placeholder*="提取码"]',
            'input[placeholder*="访问码"]',
            'input#accessCode',
            'input.vcode',
            'input[name*="code"]',
            '.pickpw input',
            '.access-code input'
        ];

        // 常见的按钮选择器
        const buttonSelectors = [
            'button[type="submit"]',
            '.g-button',
            '.pickpw button',
            'button:contains("提取")',
            'button:contains("确定")'
        ];

        let inputFound = false;

        for (const selector of inputSelectors) {
            const input = document.querySelector(selector);
            if (input && input.offsetParent !== null) { // 确保元素可见
                console.log('百度网盘助手: 找到输入框');

                // 使用第一个提取码
                const codeInfo = codes[0];
                input.value = codeInfo.code;
                input.focus();

                // 触发事件
                ['input', 'change', 'keyup'].forEach(eventType => {
                    input.dispatchEvent(new Event(eventType, { bubbles: true }));
                });

                // 查找并点击按钮
                setTimeout(() => {
                    for (const btnSelector of buttonSelectors) {
                        const button = document.querySelector(btnSelector);
                        if (button && button.offsetParent !== null &&
                            (button.textContent.includes('提取') ||
                             button.textContent.includes('确定') ||
                             button.type === 'submit')) {
                            console.log('百度网盘助手: 点击按钮');
                            button.click();
                            break;
                        }
                    }

                    // 清除使用过的提取码
                    sessionStorage.removeItem(codeInfo.key);
                }, 500);

                inputFound = true;
                break;
            }
        }

        if (!inputFound) {
            console.log('百度网盘助手: 未找到输入框，稍后重试');
        }
    }

    // 主逻辑
    if (location.hostname.includes('pan.baidu.com')) {
        // 在百度网盘页面自动填写提取码
        console.log('百度网盘助手: 百度网盘页面，启动自动填写');

        // 页面加载完成后尝试填写
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', autoFillExtractCode);
        } else {
            setTimeout(autoFillExtractCode, 300);
        }

        // 监听DOM变化
        const observer = new MutationObserver(() => {
            setTimeout(autoFillExtractCode, 100);
        });
        observer.observe(document.body, {
            childList: true,
            subtree: true
        });

    } else {
        // 在其他页面监听文本选择
        console.log('百度网盘助手: 普通页面，启动链接检测');
        handleTextSelection();
    }

})();
